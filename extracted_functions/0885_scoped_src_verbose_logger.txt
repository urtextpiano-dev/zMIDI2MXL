    pub fn scoped(self: *Self, name: []const u8) ScopedVerboseLogger {
        self.step_mutex.lock();
        const current_id = self.step_id.clone();
        self.step_mutex.unlock();
        
        return .{
            .parent = self,
            .name = name,
            .start_step = self.current_step,
            .start_step_id = current_id,
        };
    }
