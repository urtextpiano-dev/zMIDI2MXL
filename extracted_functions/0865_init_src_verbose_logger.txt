    pub fn init(allocator: std.mem.Allocator, enabled: bool, track_precision: bool) Self {

            .current_step = 0,
            .indent_level = 0,
            .start_time = @intCast(std.time.milliTimestamp()),

            .step_id = .{ .major = 0, .minor = 0, .micro = 0 },

            .step_registry = StepRegistry.init(allocator),
        };
    }
