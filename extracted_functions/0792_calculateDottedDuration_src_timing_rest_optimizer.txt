```zig
    fn calculateDottedDuration(self: *RestOptimizer, base_duration: u32, dots: u8) u32 {
        _ = self;
        var duration = base_duration;
        var dot_value = base_duration / 2;
        var i: u8 = 0;
        while (i < dots) : (i += 1) {
            duration += dot_value;
            dot_value /= 2;
        }
        return duration;
    }
```
