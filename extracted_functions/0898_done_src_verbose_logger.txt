    pub fn done(self: *const Self) void {
        if (!self.parent.enabled) return;
        
        const steps_used = self.parent.current_step - self.start_step;
        self.parent.logger.debug("{s}[{s}] Completed ({} steps)", .{ 
            self.parent.getIndent(), 
            self.name, 
            steps_used 
        });
    }
