```zig
    pub fn getInterpretedValue(self: RpnEvent) f32 {
        if (self.is_nrpn) return @floatFromInt(self.value); // Raw value for NRPN
        
        if (self.getRpnType()) |rpn_type| {
            return switch (rpn_type) {
                .pitch_bend_range => @floatFromInt(self.value >> 7), // Only MSB for semitones
                .fine_tuning => (@as(f32, @floatFromInt(self.value)) - 8192.0) / 81.92, // Cents: (value - 8192) / 81.92
                .coarse_tuning => @as(f32, @floatFromInt(self.value >> 7)) - 64.0, // Semitones: MSB - 64
                else => @floatFromInt(self.value),
            };
        }
        return @floatFromInt(self.value);
    }
```
