    pub fn init(allocator: std.mem.Allocator, ppq: u32) Quantizer {
        return .{
            .allocator = allocator,
            .ppq = ppq,
            .strength = 1.0, // Default to 100% quantization
            .grids = .{
                QuantizationGrid.init(ppq, .quarter),
                QuantizationGrid.init(ppq, .eighth),
                QuantizationGrid.init(ppq, .sixteenth),
            },
        };
    }
