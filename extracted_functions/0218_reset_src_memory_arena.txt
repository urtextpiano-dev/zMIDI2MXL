```zig
    pub fn reset(self: *ArenaAllocator) void {
        const start_time = std.time.nanoTimestamp();
        
        // Safe logger access - get fresh pointer to avoid use-after-free
        if (self.logger != null) {
            const current_logger = log_mod.getLogger();
            current_logger.debug("Arena reset: freeing {d} bytes from {d} allocations", .{
                self.stats.total_allocated,
                self.stats.allocation_count,
            });
        }
        
        _ = self.arena.reset(.free_all);
        self.stats.recordReset();
        
        const end_time = std.time.nanoTimestamp();
        const reset_time = @as(u64, @intCast(end_time - start_time));
        
        if (self.logger != null) {
            const current_logger = log_mod.getLogger();
            current_logger.debug("Arena reset completed in {d} ns", .{reset_time});
        }
    }
```
