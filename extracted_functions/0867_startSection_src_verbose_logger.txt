```zig
    pub fn startSection(self: *Self, comptime format: []const u8, args: anytype) void {
        if (!self.enabled) return;
        
        self.current_step += 1;
        const indent = self.getIndent();
        self.logger.info("{s}[STEP {:0>3}] " ++ format, .{indent, self.current_step} ++ args);
        self.indent_level += 1;
    }
```
