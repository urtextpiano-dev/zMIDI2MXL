## Function Content
```zig
    pub fn getParameterName(self: RpnEvent) []const u8 {
        if (self.is_nrpn) return "NRPN";
        if (self.getRpnType()) |rpn_type| {
            return rpn_type.getName();
        }
        return "Unknown RPN";
    }
```
