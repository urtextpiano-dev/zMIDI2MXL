## Function Content
```zig
    fn determineStaffForChord(notes: []const TimedNote) u8 {
        if (notes.len == 0) return 1; // Default to treble
        
        // Check if any note is below middle C (MIDI note 60)
        for (notes) |note| {
            if (note.note < 60) {
                return 2; // Bass staff
            }
        }
        
        return 1; // Treble staff
    }
```
