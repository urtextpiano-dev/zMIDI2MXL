```zig
    pub fn init(allocator: std.mem.Allocator) VoiceAllocator {
        var voices: [MAX_VOICES]VoiceState = undefined;
        for (0..MAX_VOICES) |i| {
            voices[i] = VoiceState{
                .number = @intCast(i + 1),
                .last_end_tick = 0,
                .note_count = 0,
            };
        }
        
        return VoiceAllocator{
            .allocator = allocator,
            .voices = voices,
            .sort_buffer = null,
        };
    }
```
