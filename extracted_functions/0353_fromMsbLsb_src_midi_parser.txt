## Function Content
```zig
    pub fn fromMsbLsb(msb: u7, lsb: u7) RpnType {
        const value: u16 = (@as(u16, msb) << 7) | lsb;
        return switch (value) {
            0x0000 => .pitch_bend_range,
            0x0001 => .fine_tuning,
            0x0002 => .coarse_tuning,
            0x3FFF => .null_rpn,
            else => @enumFromInt(value),
        };
    }
```
