```zig
    pub fn findMeasureForTick(measures: []const Measure, tick: u32) ?*const Measure {
        for (measures) |*measure| {
            if (tick >= measure.start_tick and tick < measure.end_tick) {
                return measure;
            }
        }
        return null;
    }
```
