    pub fn deinit(self: *PartInfo, allocator: std.mem.Allocator) void {
        allocator.free(self.part_id);
        allocator.free(self.part_name);
        if (self.part_abbreviation) |abbr| {
            allocator.free(abbr);
        }
        self.track_indices.deinit();
    }
