```zig
fn parseChannelPressureEvent(state: *TrackParserState, result: *TrackParseResult, status_byte: u8) (error_mod.MidiError || std.mem.Allocator.Error)!void {
    // Need 1 data byte for channel pressure events
    if (state.remainingBytes() < 1) {
        return error_mod.MidiError.UnexpectedEndOfFile;
    }
    
    const pressure = state.data[state.position];
    state.position += 1;
    
    // Validate pressure range per MIDI spec
    if (pressure > 127) {
        return error_mod.MidiError.InvalidEventData;
    }
    
    const event = ChannelPressureEvent{
        .tick = state.current_tick,
        .channel = extractChannelFromStatus(status_byte),
        .pressure = @intCast(pressure),
    };
    
    try result.channel_pressure_events.append(event);
}
```
