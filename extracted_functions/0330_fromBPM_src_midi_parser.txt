    pub fn fromBPM(tick: u32, bpm: f64) TempoEvent {
        const microseconds = @as(u32, @intFromFloat(60_000_000.0 / bpm));
        return .{
            .tick = tick,
            .microseconds_per_quarter = microseconds,
        };
    }
