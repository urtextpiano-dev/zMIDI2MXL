    pub fn findSafePoint(
        data: []const u8,
        start_pos: usize,
    ) ?usize {
        var pos = start_pos;
        
        // Look for next status byte
        while (pos < data.len) {
            if (data[pos] >= 0x80) {
                // Found a status byte
                return pos;
            }
            pos += 1;
        }
        
        return null;
    }
