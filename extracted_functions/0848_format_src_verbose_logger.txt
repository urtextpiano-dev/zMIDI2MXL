    pub fn format(self: PipelineSteps, buf: []u8) ![]const u8 {
        return std.fmt.bufPrint(buf, "[{d:0>3}.{d:0>3}.{d:0>3}]", .{
            self.getPhase(), self.getSection(), self.getStep()
        });
