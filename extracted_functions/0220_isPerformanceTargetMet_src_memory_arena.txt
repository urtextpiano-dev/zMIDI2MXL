```zig
    pub fn isPerformanceTargetMet(self: *const ArenaAllocator, target_mb: f64) bool {
        const avg_time_ns = self.stats.getAverageAllocationTime();
        const avg_time_ms = avg_time_ns / 1_000_000.0; // Convert ns to ms
        
        // Scale target based on amount of memory
        const mb_processed = @as(f64, @floatFromInt(self.stats.peak_usage)) / (1024.0 * 1024.0);
        const scaled_target_ms = if (mb_processed > 0) (target_mb / mb_processed) * 1.0 else 1.0;
        
        return avg_time_ms < scaled_target_ms;
    }
```
