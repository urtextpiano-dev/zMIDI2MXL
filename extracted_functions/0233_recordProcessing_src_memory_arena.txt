    pub fn recordProcessing(self: *EducationalPerformanceMetrics, notes_count: u64, time_ns: u64) void {
        self.notes_processed += notes_count;
        self.processing_time_per_note_ns = if (self.notes_processed > 0) 
            (self.processing_time_per_note_ns + time_ns) / 2 else time_ns;
        self.successful_cycles += 1;
    }
