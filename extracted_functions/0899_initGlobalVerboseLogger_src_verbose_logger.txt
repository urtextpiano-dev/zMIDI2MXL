```zig
pub fn initGlobalVerboseLogger(config: VerboseLoggerConfig) void {
    // Clean up existing logger if any
    if (global_verbose_logger) |*logger| {
        logger.deinit();
    }
    
    global_verbose_logger = VerboseLogger.init(
        config.allocator,
        config.enabled,
        config.track_precision,
    );
}
```
