    fn isNoteInAnyTuplet(self: *EducationalProcessor, tick: u32, tuplet_spans: []const TupletSpan) bool {
        _ = self;
        
        for (tuplet_spans) |span| {
            if (tick >= span.start_tick and tick < span.end_tick) {
                return true;
            }
        }
        return false;
    }
