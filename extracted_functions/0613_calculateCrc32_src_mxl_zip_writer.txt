```zig
    pub fn calculateCrc32(self: *const ZipWriter, data: []const u8) u32 {
        var crc: u32 = 0xFFFFFFFF;
        
        for (data) |byte| {
            const table_idx = @as(u8, @truncate(crc ^ byte));
            crc = (crc >> 8) ^ self.crc_table[table_idx];
        }
        
        return crc ^ 0xFFFFFFFF;
    }
```
